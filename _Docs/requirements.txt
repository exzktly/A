# requirements.txt
# -----------------
# Pip-only dependencies for AllWell — supports macOS (Apple Silicon) and
# Windows (x86_64). Platform-specific packages are gated by pip environment
# markers so the same file works on both.
# Python 3.10.14, tensorflow 2.13 line.
#
# ── ENVIRONMENT SETUP (macOS) ────────────────────────────────────────────────
#
#   mamba create -n allwell python=3.10.14 setuptools=69.5.1 \
#       "numpy>=1.24,<1.25" scipy scikit-image matplotlib pillow imageio h5py \
#       -c conda-forge -y
#   mamba activate allwell
#   pip install --no-build-isolation -r requirements.txt
#
# ── ENVIRONMENT SETUP (Windows) ──────────────────────────────────────────────
#
#   mamba create -n allwell python=3.10.14 setuptools=69.5.1 \
#       "numpy>=1.24,<1.25" scipy scikit-image matplotlib pillow imageio h5py \
#       pyside6 -c conda-forge -y
#   conda activate allwell
#   pip install --no-build-isolation -r requirements.txt
#
#   Notes for Windows:
#   - Native Windows TensorFlow has been CPU-only since TF 2.11. For GPU
#     acceleration on Windows, run the app under WSL2 with the Linux
#     tensorflow build + NVIDIA CUDA drivers.
#   - Installing pyside6 via conda-forge avoids needing admin to install
#     the Microsoft Visual C++ redistributable system-wide.
#
# ── COMMON ───────────────────────────────────────────────────────────────────
#
# Step 2 — verify pkg_resources works:
#
#   python -c "import pkg_resources; print('OK')"
#
# IMPORTANT:
#   - setuptools must stay at <70. Modern setuptools (v70+) removed
#     pkg_resources which is required by stardist and csbdeep.
#   - Always use --no-build-isolation with pip to prevent it from
#     escaping to the system Python for wheel builds.
#   - Never run: pip install setuptools  (this will upgrade and break things)
#
# ── VERIFY GPU AFTER INSTALL ─────────────────────────────────────────────────
#
#   python -c "import tensorflow as tf; print(tf.config.list_physical_devices('GPU'))"
#   # Expected on macOS: [PhysicalDevice(name='/physical_device:GPU:0', ...)]
#   # Expected on Windows native: []  (CPU-only — see WSL2 note above)

# ── TensorFlow ───────────────────────────────────────────────────────────────
# macOS (Apple Silicon native + Metal GPU plugin)
tensorflow-macos==2.13.0 ; sys_platform == "darwin"
tensorflow-metal==1.1.0  ; sys_platform == "darwin" and platform_machine == "arm64"
# Windows (CPU-only; TF 2.11+ dropped native Windows GPU support)
tensorflow==2.13.0       ; sys_platform == "win32"
keras==2.13.1
protobuf==3.20.3

# ── StarDist segmentation pipeline ───────────────────────────────────────────
csbdeep==0.8.0
stardist==0.9.1
numba==0.57.1
llvmlite==0.40.1

# ── Image I/O ────────────────────────────────────────────────────────────────
tifffile==2023.9.26
imagecodecs==2023.9.18

# ── GUI toolkit (Qt for Python) ──────────────────────────────────────────────
# Replaces tkinter. Pulls in shiboken6 transitively at the matching version.
# On Windows, prefer `mamba install -c conda-forge pyside6` so the MSVC
# runtime DLLs are provided by the `vc`/`vs2015_runtime` conda packages
# (avoids needing admin to install the system vcredist).
PySide6==6.7.3

# ── Build tooling (build only) ───────────────────────────────────────────────
pyinstaller==6.1.0
