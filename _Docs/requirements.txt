# requirements.txt
# -----------------
# Pip-only dependencies for AllWell — Apple Silicon macOS.
# Python 3.10.14, tensorflow-macos 2.13.0, tensorflow-metal 1.1.0
#
# ── ENVIRONMENT SETUP ────────────────────────────────────────────────────────
#
# Step 1 — create environment (setuptools pinned to <70 via mamba):
#
#   mamba create -n allwell python=3.10.14 setuptools=69.5.1 \
#       "numpy>=1.24,<1.25" scipy scikit-image matplotlib pillow imageio h5py \
#       -c conda-forge -y
#
# Step 2 — verify pkg_resources works:
#
#   mamba activate allwell
#   python -c "import pkg_resources; print('OK')"
#
# Step 3 — install pip packages:
#
#   pip install --no-build-isolation -r requirements.txt
#
# IMPORTANT:
#   - setuptools must stay at <70. Modern setuptools (v70+) removed
#     pkg_resources which is required by stardist and csbdeep.
#   - Always use --no-build-isolation with pip to prevent it from
#     escaping to the system Python for wheel builds.
#   - Never run: pip install setuptools  (this will upgrade and break things)
#
# ── VERIFY GPU AFTER INSTALL ─────────────────────────────────────────────────
#
#   python -c "import tensorflow as tf; print(tf.config.list_physical_devices('GPU'))"
#   # Expected: [PhysicalDevice(name='/physical_device:GPU:0', device_type='GPU')]
#
# ── ENVIRONMENT SETUP (Windows, native CPU) ──────────────────────────────────
#
# Native Windows TF has been CPU-only since 2.11 (TF 2.10 was the last release
# with native Windows GPU). We pin tensorflow==2.13.0 to keep the
# keras/protobuf/StarDist stack identical across platforms. For GPU on
# Windows, use WSL2 (out of scope here).
#
# Use Miniforge so `mamba` is available on Windows. From the "Miniforge Prompt":
#
#   mamba create -n allwell python=3.10.14 setuptools=69.5.1 \
#       "numpy>=1.24,<1.25" scipy scikit-image matplotlib pillow imageio h5py \
#       pyside6 -c conda-forge -y
#   mamba activate allwell
#   pip install --no-build-isolation -r requirements.txt
#
# Notes:
#   - Install pyside6 from conda-forge (brings vc/vs2015_runtime; avoids
#     needing admin rights to install vcredist).
#   - numba/llvmlite Windows wheels bundle the LLVM runtime; no separate
#     MSVC install needed for them.
#   - The same setuptools<70 / pkg_resources rationale documented above
#     applies on Windows.
#
# ── VERIFY CPU AFTER INSTALL (Windows) ───────────────────────────────────────
#
#   python -c "import tensorflow as tf; print(tf.config.list_physical_devices('GPU'))"
#   # Expected: []
#   python -c "import tensorflow as tf; print(tf.config.list_physical_devices('CPU'))"
#   # Expected: [PhysicalDevice(name='/physical_device:CPU:0', device_type='CPU')]

# ── TensorFlow (Apple Silicon, GPU via Metal) ────────────────────────────────
tensorflow-macos==2.13.0 ; sys_platform == "darwin" and platform_machine == "arm64"
tensorflow-metal==1.1.0  ; sys_platform == "darwin" and platform_machine == "arm64"

# ── TensorFlow (Windows, CPU-only) ───────────────────────────────────────────
# Native Windows TF has been CPU-only since 2.11 (TF 2.10 was the last
# release with native Windows GPU). Pinning 2.13.0 keeps keras / protobuf /
# StarDist identical across platforms. Intel macOS is not supported.
tensorflow==2.13.0 ; sys_platform == "win32"

# Shared by both TF lines (cross-platform pins)
keras==2.13.1
protobuf==3.20.3

# ── StarDist segmentation pipeline ───────────────────────────────────────────
csbdeep==0.8.0
stardist==0.9.1
numba==0.57.1
llvmlite==0.40.1

# ── Image I/O ────────────────────────────────────────────────────────────────
tifffile==2023.9.26
imagecodecs==2023.9.18

# ── Data analysis (well_viewer aggregation) ──────────────────────────────────
# pandas 2.2.3 requires numpy>=1.22.4 on py3.10 with no upper bound,
# so it resolves cleanly against the env's numpy>=1.24,<1.25 pin.
pandas==2.2.3

# ── Test environment guard ───────────────────────────────────────────────────
# exceptiongroup is a transitive dep of pytest on py<3.11. Versions >=1.3
# add `typing-extensions>=4.6.0`, which conflicts with the
# `typing-extensions<4.6.0,>=3.6.6` pin from tensorflow-macos 2.13.0 and
# silently upgrades typing-extensions, breaking TF imports. Pinning
# below 1.3 keeps pytest installable without disturbing TF.
exceptiongroup<1.3

# ── GUI toolkit (Qt for Python) ──────────────────────────────────────────────
# Replaces tkinter. Pulls in shiboken6 transitively at the matching version.
# On Windows, prefer `mamba install -c conda-forge pyside6` so the MSVC
# runtime DLLs are provided by the `vc`/`vs2015_runtime` conda packages
# (avoids needing admin to install the system vcredist).
PySide6==6.7.3

# ── macOS-only: vector PDF Copy ──────────────────────────────────────────────
# Used by well_viewer/clipboard_macos.py to write com.adobe.pdf directly to
# NSPasteboard so Copy SVG pastes as editable vector in Keynote / Pages /
# Preview. Qt6's clipboard adapter doesn't surface application/pdf to the
# macOS pasteboard, so the native path is the only way to reach iWork.
pyobjc-framework-Cocoa>=10.0 ; sys_platform == "darwin"

# ── Build tooling (build only) ───────────────────────────────────────────────
pyinstaller==6.1.0
